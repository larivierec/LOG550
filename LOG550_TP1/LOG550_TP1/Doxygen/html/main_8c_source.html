237 /*************************************************************************
238  Fonction : irq_adc_channel(void) */
239 /** @brief Interrupt Request provenant d'un EOC (End of Conversion) soit pour la lumiere ou le potentiometre
240 
241 *************************************************************************/
242 
243 __attribute__((__interrupt__))
244 static void irq_adc_channel(void)
245 {
246  // Check depassement du UART par le ADC
247  if(IS_SENSOR_VAL_RDY && IS_POTENTIOMETER_VAL_RDY)
248  {
249  booleanValues |= DEPASSEMENT_ADC;
250  }
251 
252  U32 statusRegister = AVR32_ADC.sr;
253 
254  //potentiometer conversion done
255  if(statusRegister & AVR32_ADC_IER_EOC1_MASK)
256  {
257  if(IS_POTENTIOMETER_VAL_RDY && !(AVR32_USART1.csr & (AVR32_USART_CSR_TXRDY_MASK)))
258  {
259  booleanValues |= DEPASSEMENT_UART;
260  }
261  potValue = AVR32_ADC.cdr1;
262  booleanValues |= POTENTIOMETER_VAL_RDY;
263  }
264  //light sensor conversion done
265  if(statusRegister & AVR32_ADC_IER_EOC2_MASK)
266  {
267  if(IS_SENSOR_VAL_RDY && !(AVR32_USART1.csr & (AVR32_USART_CSR_TXRDY_MASK)))
268  {
269  booleanValues |= DEPASSEMENT_UART;
270  }
271  lightSensorValue = AVR32_ADC.cdr2;
272  booleanValues |= SENSOR_VAL_RDY;
273  }
274  CLEAR_ADC_CONVERSION_FLAG;
275 }

299 
300 void adc_init(void)
301 {
302  pcl_switch_to_osc(PCL_OSC0, FOSC0, OSC0_STARTUP);
303  //Configure the ADC to use 8 bit values instead of 10
304  AVR32_ADC.mr |= 1 << AVR32_ADC_LOWRES_OFFSET;
305 
306  //Enable the Light Sensor
307  AVR32_ADC.cher = 1 << LIGHT_CHANNEL;
308  //Enable the potentiometer
309  AVR32_ADC.cher = 1 << POT_CHANNEL;
310 
311 
312  INTC_register_interrupt(&irq_adc_channel,AVR32_ADC_IRQ,P_HIGH);
313  AVR32_ADC.ier = AVR32_ADC_IER_EOC2_MASK | AVR32_ADC_IER_EOC1_MASK;
314 }
