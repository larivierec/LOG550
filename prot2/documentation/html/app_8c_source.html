114 int main (void)
115 {
116  BSP_Init(); /*Set le timer et demarre les ISR_TICKS.*/
117  CPU_IntDis(); /*Desactive toute les interrupts pendant l'initialisation */
118  OSInit(); /*Initialise "uC/OS-II, The Real-Time Kernel" */
119  Init_IO_Usager(); /*Initialisation des differents I/O par l'usager */
120 
121  /*Creation de toute les taches, a des priorites differentes (1<prio<25) */
122  OSTaskCreate(LED_flash, NULL, (OS_STK *)&LED_Stack[OS_TASK_STK_SIZE-1], 1);
123  OSTaskCreate(UART_Cmd_RX, NULL, (OS_STK *)&UART_RX_Stack[OS_TASK_STK_SIZE-1], 2);
124  OSTaskCreate(UART_SendSample, NULL, (OS_STK *)&UART_TX_Stack[OS_TASK_STK_SIZE-1], 3);
125  OSTaskCreate(ADC_Cmd, NULL, (OS_STK *)&ADC_Stack[OS_TASK_STK_SIZE-1], 4);
126  OSTaskCreate(Alarm_msgQ, NULL,(OS_STK *)&Alarm_Stack[OS_TASK_STK_SIZE-1], 5);
127  OSTaskCreate(Bonus_Thread, NULL,(OS_STK *)&Bonus_Stack[OS_TASK_STK_SIZE-1], 6);
128 
129 
130  Start_Sem = OSSemCreate(0); /*Creation du semaphore Start_Sem*/
131  UART_TX_Sem = OSSemCreate(0); /*Creation du semaphore UART_TX_Sem*/
132  Alarm_Sem = OSSemCreate(0); /*Creation du semaphore Alarm_Sem*/
133  LED1_Sem = OSSemCreate(0); /*Creation du semaphore LED1_Sem*/
134  SEM_ADC_Alternate = OSSemCreate(0); /*Creation du semaphore SEM_ADC_Alternate*/
135  ADCMsg = OSQCreate(ADCMsgFIFO, 4); /*Creation du message queue de taille 4 ADCMsg */
136 
137  OSStart();
138  return (0);
139 }

225 static void ADC_Cmd (void *p_arg)
226 {
227  (void)p_arg;
228 
229  while (1) {
230  if(ADCMsg->OSEventCnt >= 4)
231  {
232  OSSemPost(Alarm_Sem);
233  }
234 
235  if(Start_Sem->OSEventCnt == 1)
236  {
237  AVR32_ADC.cr = AVR32_ADC_START_MASK; // starts the ADC Conversion
238  // This is just to get a visual queue. Must delete.
239  CPU_INT32U statusRegister = AVR32_ADC.sr;
240  if((statusRegister & AVR32_ADC_IER_EOC1_MASK) && SEM_ADC_Alternate->OSEventCnt == 0)
241  {
242  OSSemSet(SEM_ADC_Alternate,1,NULL);
243  OSQPost(ADCMsg, (AVR32_ADC.cdr1 | 0x01));
244  }
245 
246  if((statusRegister & AVR32_ADC_IER_EOC2_MASK) && SEM_ADC_Alternate->OSEventCnt == 1)
247  {
248  OSSemSet(SEM_ADC_Alternate,0,NULL);
249  OSQPost(ADCMsg, (AVR32_ADC.cdr2 & ~0x01));
250  }
251  }
252  OSTimeDly(1); // Delai en TICKS (1 TICKS=1milisec)
253  }
254 }
